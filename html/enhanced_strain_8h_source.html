215 template <int dim>
216 Tensor<1,dim,double> enhancedStrain<dim>::tilde_grad(unsigned int function_no, unsigned int point_no){
217  //Enhanced strain gradient functions for the four incompatible shape functions
218 
219  //Ensure the initiliaziation function has been called
220  if(enh_dofs_initialized == false){
221  *pcout << "Error: enhanced dofs not yet initialized.\n";
222  exit(1);
223  }
224 
225  //Tensor holding values of the specified enhanced strain gradient function at the given quad. point
226  Tensor<1,dim,double> grad; grad = 0.;
227  //Jacobian at quadrature point
228  Tensor<2,dim,double> J_xi = fe_values.jacobian(point_no);
229  double j_xi = determinant(J_xi);
230 
231  //Get the coordinates of the quadrature point in the unit domain
232  Point<dim,double> xi = fe_values.get_quadrature().point(point_no);
233  //Convert xi to the bi-unit domain (deal.II uses a unit domain)
234  for(unsigned int i=0; i<dim; i++){
235  xi(i) = 2.*xi(i) - 1.;
236  }
237 
238  //Define enhanced shape function gradients (see paper cited at top)
239  if(function_no < 3*dim){
240  //\tilde{N^i} = 0.5*(xi(i)^2 - 1), i = 0,1,2, from functions listed
241  //in eqn (3.8) of the cited paper
242  grad[function_no/dim] = xi(function_no/dim);
243  }
244  else if(function_no >= 3*dim && function_no < 4*dim){
245  //\tilde{N^3} = xi(0)*xi(1)*xi(2), see equation (3.17) from cited paper
246  grad[0] = xi(1)*xi(2);
247  grad[1] = xi(0)*xi(2);
248  grad[2] = xi(0)*xi(1);
249  }
250  else{
251  *pcout << "Error: invalid function number.\n";
252  exit(1);
253  }
254 
255  //See the 2nd line of equation (4.1) of the cited paper
256  grad = j_0/j_xi*grad*J_0inv;
257 
258  return grad;
259 }

261 template <int dim>
262 Tensor<1,dim,double> enhancedStrain<dim>::bar_grad(unsigned int function_no, unsigned int point_no,Tensor<1,dim,double> alpha_4){
263  //Enhanced strain gradient functions to replace the standard basis function gradients
264 
265  if(enh_dofs_initialized == false){
266  *pcout << "Error: enhanced dofs not yet initialized.\n";
267  exit(1);
268  }
269 
270  //Hold enhanced strain gradient function value
271  Tensor<1,dim,double> grad;
272  //Jacobian at given quad. point
273  Tensor<2,dim> J_xi = fe_values.jacobian(point_no);
274  double j_xi = determinant(J_xi);
275 
276  //Modified basis function gradient to replace standard basis function gradient
277  //See line 3 of equation (4.1) of paper cited at top, as well as (2.17).
278  //Note that deal.II's basis function gradients are all w.r.t. the actual domain,
279  //not the bi-unit or unit domain.
280  grad = (1. + alpha_4*tilde_grad(3*dim,point_no))*center_grad(function_no);
281  grad += j_0/j_xi*(fe_values.shape_grad(function_no,point_no) - center_grad(function_no))*J_xi*J_0inv;
282 
283  return grad;
284 }

311 template <int dim>
312 void enhancedStrain<dim>::get_F_enh(unsigned int q, FullMatrix<double> &F){
313 
314  if(enh_dofs_initialized == false){
315  *pcout << "Error: enhanced dofs not yet initialized.\n";
316  exit(1);
317  }
318 
319  //Get the number of standard dofs and primary enhanced dofs per element
320  unsigned int dofs_per_elem = fe_values.get_fe().dofs_per_cell;
321  unsigned int prim_enh_dofs_per_elem = primary_enhanced_dofs_per_cell();
322 
323  //Create vector of original location of each standard dof in the element
324  //This location is the x, y, or z location based on the orientation of the dof.
325  MappingQ1<dim,dim> mapping;
326  std::vector<double> nodalCoords (dofs_per_elem);
327  for (unsigned int d=0; d<dofs_per_elem; ++d){
328  //Check the orientation of the standard dof (displacement in the x, y, or z direction)
329  unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
330  //Get the corresponding component of the position of the node in the initial configuration,
331  //i.e. if the dof is displacement in the y direction at a particular node, store the y component
332  //of that node's location.
333  nodalCoords[d] = mapping.transform_unit_to_real_cell(currentElem,fe_values.get_fe().unit_support_point(d))[i];
334  }
335 
336  //define F_tilde (based on enhanced dofs)
337  //see the 2nd half of equation (4.2) of the paper
338  Tensor<2,dim,double> F_tilde; F_tilde = 0.0;
339  for (unsigned int A=0; A<prim_enh_dofs_per_elem; ++A){
340  unsigned int i = enhanced_system_to_component_index(A);
341  F_tilde[i] += Alpha(4*dim*currentElem->user_index()+A)*tilde_grad(A,q);
342  }
343 
344  //The fourth alpha vector
345  Tensor<1,dim,double> alpha_4;
346  for (unsigned int i=0; i<dim; ++i){
347  alpha_4[i] = Alpha(4*dim*currentElem->user_index()+9+i);
348  }
349 
350  //define F_bar, F_0 (based on standard dofs)
351  //see the 1st half of equation (4.2) of the paper
352  Tensor<2,dim,double> F_bar; F_bar = 0.0;
353  for (unsigned int d=0; d<dofs_per_elem; ++d){
354  unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
355  //F is the derivative of (displacement + original position = deformed position)
356  F_bar[i] += (Ulocal[d] + nodalCoords[d])*bar_grad(d,q,alpha_4);
357  }
358 
359  //define F_enhanced = GRADbar_{X}[phi] + GRADtilde_{X}[phi]
360  //equation (4.2) of the paper
361  Tensor<2,dim,double> F_enh; F_enh = 0.0;
362  F_enh = F_bar + F_tilde;
363 
364  //Copy F_enh into the user's F matrix
365  F.copy_from(F_enh);
366 }

368 template <int dim>
369 void enhancedStrain<dim>::create_block_mat_vec(FullMatrix<double> F, FullMatrix<double> tau, Tensor<4,dim,double> c_ep, unsigned int q){
370  //Refer to the paper cited at the top
371 
372  if(enh_dofs_initialized == false){
373  *pcout << "Error: enhanced dofs not yet initialized.\n";
374  exit(1);
375  }
376 
377  Klocal = 0.; Glocal = 0.; Mlocal = 0.;
378  Flocal = 0.; Hlocal = 0.;
379 
380  Tensor<2,dim,double> F_enh, Fenh_inv, tau_enh;
381  unsigned int dofs_per_elem = fe_values.get_fe().dofs_per_cell;
382  unsigned int prim_enh_dofs_per_elem = primary_enhanced_dofs_per_cell();
383  unsigned int enh_dofs_per_elem = enhanced_dofs_per_cell();
384 
385  //Copy inputs to the class variables
386  F.copy_to(F_enh);
387  tau.copy_to(tau_enh);
388  Fenh_inv = invert(F_enh);
389 
390  //Retrieve the 4th Alpha vector for this element
391  Tensor<1,dim,double> alpha_4;
392  for (unsigned int i=0; i<dim; ++i){
393  alpha_4[i] = Alpha(4*dim*currentElem->user_index()+9+i);
394  }
395 
396  //Create vector of original location of each standard dof in the element
397  //This location is the x, y, or z location based on the orientation of the dof.
398  MappingQ1<dim,dim> mapping;
399  std::vector<double> nodalCoords (dofs_per_elem);
400  Tensor<2,dim,double> F_0; F_0 = 0.0;
401  for (unsigned int d=0; d<dofs_per_elem; ++d){
402  //Check the orientation of the standard dof (displacement in the x, y, or z direction)
403  unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
404  //Get the corresponding component of the position of the node in the initial configuration,
405  //i.e. if the dof is displacement in the y direction at a particular node, store the y component
406  //of that node's location.
407  nodalCoords[d] = mapping.transform_unit_to_real_cell(currentElem,fe_values.get_fe().unit_support_point(d))[i];
408  //F is the derivative of (displacement + original position = deformed position)
409  F_0[i] += (Ulocal[d] + nodalCoords[d])*center_grad(d);
410  }
411 
412  //Transform enhanced gradients to current configuration
413  //see equation (4.3) of the cited paper
414  std::vector<Tensor<1,dim,double> > bar_del(dofs_per_elem),
415  center_del(dofs_per_elem),
416  tilde_del(prim_enh_dofs_per_elem);
417  for (unsigned int d=0; d<dofs_per_elem; ++d){
418  bar_del[d] = bar_grad(d,q,alpha_4)*Fenh_inv;
419  center_del[d] = center_grad(d)*Fenh_inv;
420  }
421  for (unsigned int d=0; d<prim_enh_dofs_per_elem; d++){
422  tilde_del[d] = tilde_grad(d,q)*Fenh_inv;
423  }
424  Tensor<2,dim,double> F0_Finv;
425  F0_Finv = F_0*Fenh_inv;
426 
427  //Evaluate H = F_intTilde_elem
428  for (unsigned int d=0; d<enh_dofs_per_elem; d++){
429  unsigned int i = enhanced_system_to_component_index(d);
430  if(d<prim_enh_dofs_per_elem){
431  //see line 2 of equation (4.9)
432  Hlocal(d) += tau_enh[i]*tilde_del[d];
433  }
434  else{
435  //see line 2 of equation (4.9)
436  Hlocal(d) += double_contract(F0_Finv,tau_enh)*tilde_grad(d,q)[i];
437  }
438  }
439 
440  //Evaluate F = F_intBar_elem
441  for (unsigned int d=0; d<dofs_per_elem; ++d){
442  unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
443  //see line 1 of equation (4.9)
444  Flocal(d) += tau_enh[i]*bar_del[d];
445  }
446 
447  //Calculate M (material and geometric parts)
448  for (unsigned int I=0; I<enh_dofs_per_elem; ++I){
449  unsigned int j = enhanced_system_to_component_index(I);
450  for (unsigned int J=0; J<enh_dofs_per_elem; ++J){
451  unsigned int k = enhanced_system_to_component_index(J);
452  for (unsigned int i=0; i<dim; ++i){
453  for (unsigned int l=0; l<dim; ++l){
454  if(I < prim_enh_dofs_per_elem && J < prim_enh_dofs_per_elem){
455  //See line 3 of equations (4.14) and (4.18)
456  Mlocal[I][J] += tilde_del[I][i]*(tau_enh[i][l]*(j==k) + c_ep[i][j][k][l])*tilde_del[J][l];
457  }
458  else if(I < prim_enh_dofs_per_elem){
459  //See also line 2 of equation (4.19)
460  Mlocal[I][J] += (tilde_del[I][i]*tau_enh[i][l]*F0_Finv[j][l] + F0_Finv[i][l]*c_ep[i][l][j]*tilde_del[I])*
461  tilde_grad(J,q)[k];
462  }
463  }
464  if(I < prim_enh_dofs_per_elem && J >= prim_enh_dofs_per_elem){
465  Mlocal[J][I] = Mlocal[I][J];
466  }
467  }
468  //See also line 3 of equation (4.19)
469  if(I >= prim_enh_dofs_per_elem && J >= prim_enh_dofs_per_elem){
470  Mlocal[I][J] += (contract3(F0_Finv,c_ep,F0_Finv) + double_contract(F0_Finv,tau_enh*F0_Finv))*
471  tilde_grad(I,q)[j]*tilde_grad(J,q)[k];
472  }
473  }
474  }
475 
476  //Fill in local K matrix (both material and geometric parts)
477  for (unsigned int I=0; I<dofs_per_elem; ++I){
478  unsigned int j = fe_values.get_fe().system_to_component_index(I).first;
479  for (unsigned int J=0; J<dofs_per_elem; ++J){
480  unsigned int k = fe_values.get_fe().system_to_component_index(J).first;
481  for (unsigned int i=0; i<dim; ++i){
482  for (unsigned int l=0; l<dim; ++l){
483  //See line 1 of equations (4.14) and (4.18)
484  Klocal[I][J] += bar_del[I][i]*(c_ep[i][j][k][l] + tau_enh[i][l]*(j==k))*bar_del[J][l];
485  }
486  }
487  }
488  }
489 
490  //Fill in local G matrix (both material and geometric parts)
491  for (unsigned int I=0; I<dofs_per_elem; ++I){
492  unsigned int j = fe_values.get_fe().system_to_component_index(I).first;
493  for (unsigned int J=0; J<enh_dofs_per_elem; ++J){
494  unsigned int k = enhanced_system_to_component_index(J);
495  for (unsigned int i=0; i<dim; ++i){
496  for (unsigned int l=0; l<dim; ++l){
497  if(J<prim_enh_dofs_per_elem){
498  //See line 2 of equations (4.14) and (4.18)
499  Glocal[I][J] += bar_del[I][i]*(c_ep[i][j][k][l] + tau_enh[i][l]*(j==k))*tilde_del[J][l];
500  }
501  else{
502  //See also line 1 of equation (4.19)
503  Glocal[I][J] += (F0_Finv[i][l]*c_ep[i][l][j]*bar_del[I] + (F0_Finv[j][i]*bar_del[I][l] + (j==i)*center_del[I][l])*tau_enh[i][l])*
504  tilde_grad(J,q)[k];
505  }
506  }
507  }
508  }
509  }

516

520  }

524  for (unsigned int n=0; n<dofs_per_elem; ++n){
525  staticCondensationData[cellID].d(n) = dUlocal[n];

534  }
