533  strain_energy.grad(varStrainEnergy,2+A)*varStrainEnergy[2+A]*(A==B);
534  }
535  }
536  if(yield_TR <= 0){
537  //If elastic...
538  a_ep = a_e;
539  }
540  else{
541  //If plastic...
542 
543  //The next several lines go into equation (35)
544  FullMatrix<double> a_eInv(dim,dim), f2_B(dim,dim);
545  a_eInv.invert(a_e);
546  //Find the 2nd derivatives of f w.r.t. beta
547  for (unsigned int A=0; A<dim; ++A){
548  for (unsigned int B=0; B<dim; ++B){
549  f2_B[A][B] = yield.hess(varYield,A,B);
550  }
551  }
552 
553  FullMatrix<double> h1(dim,dim), h2(dim,dim), h3(dim,dim), h3h2(dim,dim), h2h3h2(dim,dim), h2_etc(dim,dim);
554  double h4;
555  //Find h1
556  h1.equ(1,Identity,2./3.*properties.H*gamma_Dt,f2_B);
557  h1.gauss_jordan();
558  //Find h2
559  h1.mmult(h2,f2_B); //h1*f2_B
560  h2.equ(1,Identity,-2./3.*properties.H*gamma_Dt,h2);
561  //Find h3
562  f2_B.mmult(h3,h2); //f2_B*h2
563  h3.equ(1.,a_eInv,gamma_Dt,h3);
564  h3.gauss_jordan();
565  //Find h4
566  h4 = 1. - gamma_Dt*harden.grad(varIsoHardening,0)*yield.hess(varYield,4,4);
567  //Find h3*h2
568  h3.mmult(h3h2,h2);
569  //Find h2*h3*h2
570  h2.mmult(h2h3h2,h3h2);
571  //Find h2*h3*h2 + 2/3*H*h1
572  h2_etc.equ(1,h2h3h2,2./3.*properties.H,h1);
573 
574  Vector<double> h3h2_nuBar(dim), h2etc_nuBar(dim);
575  //Temporary variables, recall that nuBar is the partial of f w.r.t. beta
576  h3h2.vmult(h3h2_nuBar,nuBar);
577  h2_etc.vmult(h2etc_nuBar,nuBar);
578 
579  //The next two lines are equation (34)
580  a_ep.outer_product(h3h2_nuBar,h3h2_nuBar);
581  a_ep.equ(1.,h3,-h4/(h4*(nuBar*h2etc_nuBar) - pow(yield.grad(varYield,4),2.)*harden.grad(varIsoHardening,0)),a_ep);
582 
583  }
584  for (unsigned int i=0; i<dim; ++i){
585  for (unsigned int j=0; j<dim; ++j){
586  for (unsigned int k=0; k<dim; ++k){
587  for (unsigned int l=0; l<dim; ++l){
588  if(this->currentIncrement == 0 && this->currentIteration==0){
589  //For readability, extract the Lame parameters
590  double lambda = varStrainEnergy[0], mu = varStrainEnergy[1];
591  //c_{ijkl}=C_{IJKL}, the standard modulus, in the limit as stretches go to one
592  //This is done to avoid dividing by zero.
593  c[i][j][k][l] = lambda*(i==j)*(k==l) + mu*((i==k)*(j==l) + (i==l)*(j==k));
594  }
595  else{
596  //These two for loops are computing equations (33) and (38) of the formulation
597  for (unsigned int A=0; A<dim; ++A){
598  for (unsigned int B=0; B<dim; ++B){
599  c[i][j][k][l] += a_ep[A][B]*eigDyad[A][i][j]*eigDyad[B][k][l];
600  }
601  c[i][j][k][l] += 2.*beta(A)/d_A(A)*(0.5*(b_eTR[i][k]*b_eTR[j][l] + b_eTR[i][l]*b_eTR[j][k])
602  - b_eTR[i][j]*b_eTR[k][l]
603  - b_eTR.determinant()/eigTR(A)*(0.5*((i==k)*(j==l) + (i==l)*(j==k))
604  - ((i==j) - eigDyad[A][i][j])*((k==l) - eigDyad[A][k][l]))
605  + eigTR(A)*(b_eTR[i][j]*eigDyad[A][k][l]
606  + eigDyad[A][i][j]*b_eTR[k][l]
607  + (b_eTR.trace() - 4.*eigTR(A))*eigDyad[A][i][j]*eigDyad[A][k][l]));
608  }
609  }
610  }
611  }
612  }
613  }
